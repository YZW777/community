  1. /**
  2.  * 封装分页相关的信息.

  4. public class Page {
  5.  
  6.     // 当前页码

  8.     // 显示上限

  11.     private int rows;
  12.     // 查询路径(用于复用分页链接)
  13.     private String path;

  15.     public int getCurrent() {

  24.  
  25.     public int getLimit() {
  26.         return limit;
  27.     }
  28.  
  29.     public void setLimit(int limit) {

  31.             this.limit = limit;

  33.     }
  34.  
  35.     public int getRows() {
  36.         return rows;

  43.     }

  48.

  52.  
  53.     /**
  54.      * 获取当前页的起始行
  55.      *
  56.      * @return
  57.      */
  58.     public int getOffset() {

  60.         return (current - 1) * limit;

  62.  
  63.     /**
  64.      * 获取总页数
  65.      *
  66.      * @return
  67.      */
  68.     public int getTotal() {
  69.         // rows / limit [+1]
  70.         if (rows % limit == 0) {
  71.             return rows / limit;
  72.         } else {
  73.             return rows / limit + 1;
  74.         }
  75.     }
  76.  
  77.     /**
  78.      * 获取起始页码
  79.      *
  80.      * @return

  82.     public int getFrom() {

  92.     public int getTo() {
